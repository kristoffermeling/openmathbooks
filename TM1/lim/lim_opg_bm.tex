\op{opglimx}
For hvilke $ x $-verdier er $ f(x)=\frac{|x|}{x} $ kontinuerlig?

\op{opglimkont}
Undersøk om funksjonene er kontinuerlige for $ x=4 $.
\begin{equation*}
	f(x)= \left\lbrace{
		\begin{array}{rcr}
			x-5 &,&x<4 \br
			-x+3   &,& x\geq 4
		\end{array}
	}\right. 
\end{equation*}

\begin{equation*}
	g(x)= \left\lbrace{
		\begin{array}{rcr}
			x^2+7 &,&x<4 \br
			5x+1   &,& x\geq 4
		\end{array}
	}\right. 
\end{equation*}

\begin{equation*}
	h(x)= \left\lbrace{
		\begin{array}{rcr}
			2x^2-4 &,&x>4 \br
			-\frac{1}{2}x+30   &,& x\leq 4
		\end{array}
	}\right. 
\end{equation*}

\op{opglimfinna}
Bestem $ a $ slik at $ f $ er kontinuerlig
\begin{equation*}
	f(x)= \left\lbrace{
		\begin{array}{rcr}
			x^2 - 4 + a &,& x < 3 \\
			2x + 1 & ,& x \geq 3
		\end{array}
	}\right. 
\end{equation*}
